    /**
     * Checks if the inserted code  belongs to the same country as the target code.
     * Generally this only makes senses if one code is alpha-2 and the other is alpha-3
     * @param insertedCode alpha-2 or alpha-3 or numeric country code
     * @param targetCode alpha-2 or alpha-3 or numeric country code
     */
    static insertCodeIsGivenCountry(insertedCode: string, targetCode: string): boolean {
 
        const code1 = insertedCode.toUpperCase();
        const code2 = targetCode.toUpperCase();
 
        if (code1 === code2) {

        }
 
        const countryCode = countryCodes.find(country =>

        return Boolean(countryCode);
    }
