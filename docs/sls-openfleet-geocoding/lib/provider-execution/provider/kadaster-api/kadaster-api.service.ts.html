    // Building the Kadaster API search string to search for the basic address details via the address parameters
    protected buildKadasterAPIAddressSearchString(clientSettings: ClientGeocodingSettings, decodedAddress: DecodedAddress)
        : string {
 
        let searchString = '';
 
        let searchAddressComponents = {
            houseNumber: true,
            houseName: true,
            address1: true,
            address2: true,
            address3: true,
            address4: true,
            postCode: true,
            locality: true,
            region: true,
            countryCode: true
        };
 
        if (clientSettings && clientSettings.searchAddressComponents) {
            searchAddressComponents = clientSettings.searchAddressComponents;
        }
 
        if (searchAddressComponents.address1 && decodedAddress.address1) {
            // Replace slash with encoded slash and dash with encoded dash
            let newAddress1 = decodedAddress.address1.replace(/\//g, '%2F');
            newAddress1 = newAddress1.replace(/-/g, '%2D');
            newAddress1 = newAddress1.replace(/#/g, '');
 
            searchString += 'straat=' + newAddress1;
 
        }
        if (searchAddressComponents.houseNumber && decodedAddress.houseNumber) {
            // Replace slash with encoded slash and delete any #
            let newHousenumber = decodedAddress.houseNumber.replace(/\//g, '%2F');
            newHousenumber = newHousenumber.replace(/#/g, '');
 
            const splitHouseNumber = newHousenumber.split('-');
 
            const specialCharRegex = /[ `!@#$%^&*()_+\-=\[\]{};':"\\|,.<>\/?~]/;
            const numberRegex = /^\d+$/;
 
            if (splitHouseNumber.length > 1 && splitHouseNumber[1].length === 1 &&
                !specialCharRegex.test(splitHouseNumber[1]) && !numberRegex.test(splitHouseNumber[1])) {
 
                searchString += '&huisletter=' + splitHouseNumber[1];
            }
 
            searchString += '&huisnummer=' + splitHouseNumber[0];
 
        }
 
        if (searchAddressComponents.postCode && decodedAddress.postCode) {
            if (clientSettings.postcodeExceptions && clientSettings.postcodeExceptions.postcodesToRemove) {
                if (!clientSettings.postcodeExceptions.postcodesToRemove.includes(decodedAddress.postCode)) {
                    searchString += '&postcode=' + decodedAddress.postCode.replace(' ', '');
                }
            }
            else {
                searchString += '&postcode=' + decodedAddress.postCode.replace(' ', '');
            }
        }
        if (searchAddressComponents.locality && decodedAddress.locality) {
 
            let newlocality = decodedAddress.locality;
            newlocality = newlocality.replace(/#/g, '');
            searchString += '&woonplaats=' + newlocality;
        }
 
        searchString = searchString.replace('/', ' ');
 
        return searchString;
    }
 
    protected getResultDetails(detailedResponse: KadasterDetailedResponse, countryFormat: string)
        : { formattedAddress: string; detailedAddress: DetailedAddress; } {
 
        const isoOption = CountryCodeUtils.getIsoCode(detailedResponse.land, countryFormat);
        const isoCountryCode = isoOption.isPresent() ? isoOption.get() : detailedResponse.land;
 
        const clean = (val: any): string =>
            typeof val === 'string' && (val.trim() === '' || val.trim() === 'undefined' || val.trim() === 'null')
                ? ''
                : (val || '').toString().trim();
 
        const postcode = clean(detailedResponse.postcode).replace(/\s+/g, '');
        const huisnummer = clean(detailedResponse.huisnummer);
        const huisnummertoevoeging = clean(detailedResponse.huisnummertoevoeging);
        const houseNumber = huisnummer && huisnummertoevoeging ? `${huisnummer}-${huisnummertoevoeging}`
            : huisnummer || huisnummertoevoeging || '';
 
        const street = clean(detailedResponse.straat);
        const subCommune = clean(detailedResponse.woonplaats);
        const locality = clean(detailedResponse.buurt_naam);
        const region = clean(detailedResponse.wijk_naam);
        const countryCode = clean(isoCountryCode);
 
        const detailedAddress: DetailedAddress = {
            postcode,
            locality,
            subCommune,
            region,
            street,
            houseNumber,
            countryCode
        };
 
        let formattedAddress = '';
 
        if (street) {
            const streetPart = houseNumber ? `${street} ${houseNumber}` : street;
 
            const parts = [
                streetPart,
                postcode,
                subCommune,
                locality,
                region,
                countryCode,
            ];
 
            formattedAddress = parts.filter(p => p && p.trim().length > 0).join(', ');
        } else {
            const parts = [
                `${postcode} ${locality}`.trim(),
                region,
                countryCode,
            ];
 
            formattedAddress = parts.filter(p => p && p.trim().length > 0).join(', ');
        }
 
        return { formattedAddress, detailedAddress };
    }
 
    /**
     * Reports if the kadaster address is valid.
     * @returns 
     */
    protected validateKadasterResult(requestAddress: DecodedAddress, result: KadasterSearchResponse,
        _clientSettings: ClientGeocodingSettings, postcodeCharRemoval: string[] | undefined): {
            valid: boolean,
            potential: boolean,
            reason?: string
        } {
 
        const streetValidations = _clientSettings.addressValidation?.streetValidation;
 
        if (requestAddress.street && result?.straat && streetValidations) {
            for (const validation of streetValidations) {
                const isInEntry = result.straat.toUpperCase().includes(validation);
                const isInDecoded = requestAddress.street.includes(validation);
 
                // If validation string is found in only one of the other fields, return false
                if ((isInEntry && !isInDecoded) || (!isInEntry && isInDecoded)) {
                    return { valid: false, potential: false, reason: `Rejected due to street validation rules` };
                }
            }
        }
 
        if (result?.postcode && requestAddress.postCode) {
 
            let reqPostcode = requestAddress.postCode.replace(/\s/g, '');
 
            if (postcodeCharRemoval) {
                for (const pcChar of postcodeCharRemoval) {
 
                    // Handling cases where all letters are to be removed from the postcodes
                    if (pcChar === 'letters') {
                        result.postcode = result.postcode.replace(/\D/g, '');
                        reqPostcode = reqPostcode.replace(/\D/g, '');
                    }
                    else {
                        result.postcode = result.postcode.replace(pcChar, '');
                        reqPostcode = reqPostcode.replace(pcChar, '');
                    }
                }
            }
 
            // Applicable for clients who require a postcode prefix check
            let postCodePrefix = _clientSettings.postcodePrefixCheck;
 
            if (_clientSettings.addressValidation && _clientSettings.addressValidation.postcodeSettings && reqPostcode.length !== 6) {
                for (const postcodeSettings of _clientSettings.addressValidation.postcodeSettings) {
 
                    // Retrieve the settings for the particular country code that requires these settings
                    if (postcodeSettings.countryCode === requestAddress.countryCode || postcodeSettings.countryCode === 'ALL_CODES') {
 
                        // Setting the prefix check here since this is countryCode specific and not client
                        if (postcodeSettings.postcodePrefixCheck) {
                            postCodePrefix = postcodeSettings.postcodePrefixCheck;
                        }
 
                        if (postcodeSettings.checkPostcodeHouseNumValidation) {
 
                            let requestHouseNumber = requestAddress.houseNumber ? requestAddress.houseNumber.replace(/\D/g, '') : '';
                            const responseHouseNumber = result.huisnummer ? result.huisnummer.replace(/\D/g, '') : '';
 
                            // Handling cases where the matching house number would be sent in the street field
                            if (requestHouseNumber === '' && requestAddress.address1
                                && requestAddress.address1.indexOf(responseHouseNumber) > -1) {
                                requestHouseNumber = responseHouseNumber;
                            }
 
                            // Logic to cater for cases where if street and city are missing from request
                            // then only return match if house number is matching
                            if ((!requestAddress.address1 || requestAddress.address1 === '') &&
                                requestHouseNumber !== responseHouseNumber) {
                                return { valid: false, potential: false, reason: `Empty street and non matching house number` };
                            }
 
                            // Logic to cater for cases street and house number are returned
                            if (result.straat && requestAddress.address1 &&
                                (result.straat.trim().toUpperCase().indexOf(requestAddress.address1?.trim().toUpperCase()) < 0)
                                && requestHouseNumber !== responseHouseNumber) {
                                return { valid: false, potential: false, reason: `Non matching street and house number` };
                            }
 
                            // Logic to cater for when postcode doesn't match but street, house number and city match
                            // In this case accept the returned postcode as the correct postcode
                            if (requestAddress.address1 && reqPostcode !== result.postcode
                                && requestHouseNumber === responseHouseNumber
                                && (requestAddress.address1?.toUpperCase().includes(result.straat?.toUpperCase()) ||
                                    result.straat?.toUpperCase().includes(requestAddress.address1?.toUpperCase()))
                                && requestAddress.city?.toUpperCase() === result.woonplaats?.toUpperCase()) {
                                requestAddress.postCode = result.postcode;
                                reqPostcode = result.postcode;
                            }
                        }
                    }
                }
            }
 
            if (reqPostcode.length === 6) {
                postCodePrefix = 6;
            }
 
            if (postCodePrefix) {
 
                const originalPostcode = reqPostcode.substring(0, postCodePrefix);
                const returnedPostcode = result.postcode.substring(0, postCodePrefix);
 
                if (originalPostcode !== returnedPostcode) {
                    return { valid: false, potential: false, reason: `Non matching postcodes` };
                }
            }
        }
        else {
            return { valid: false, potential: false, reason: `Result has missing postcode` };
        }
 
        return { valid: true, potential: true, reason: `Successful Validation` };
    }

    async geocodeAddress(clientSettings: ClientGeocodingSettings,
        executionDefinition: ProviderExecutionDefinition, decodedAddress: DecodedAddress): Promise<GeocodingResult> {
        logger.info('Called Kadaster API');
        const apiKey = executionDefinition.providerSettings.options.get('apiKey');
        const sourceEndPoint = executionDefinition.providerSettings.options.get('sourceEndPoint');
        const defaultRooftopScore = executionDefinition.providerSettings.options.get('defaultRooftopScore') as number | undefined;
 
        // List of characters to be removed from postcodes for specific clients
        const postcodeCharRemoval = executionDefinition.providerSettings.options.get('postcodeCharRemoval') as string[] | undefined;
 
        const countryFormat = clientSettings.countryCodeFormat || countryCodeFormat.Alpha2;
 
        const providerType = executionDefinition.providerSettings.providerType;
 
        if (apiKey === undefined || sourceEndPoint === undefined) {
            return new GeocodingResult(Result.Failed, Reason.MissingProviderOptions, ProviderKey.KadasterAPI,
                providerType, `One of 'apiKey' or 'sourceEndPoint was not defined in provider settings!`);
        }
 
        // Checking if the client has a limitation on the street input - char length or no non-numerical values
        if (clientSettings.streetCharCheck && decodedAddress.address1) {
            // Checking character length and whether string contains letters
            Eif (decodedAddress.address1.length < clientSettings.streetCharCheck || !/[a-zA-Z]/g.test(decodedAddress.address1)) {
                return new GeocodingResult(Result.Failed, Reason.NotMatchingParams, ProviderKey.KadasterAPI,
                    providerType, `Street name invalid (Under 3 characters or only contains numerical values)`);
            }
        }
 
        // Rejects addresses with missing house numbers
        if (!decodedAddress.houseNumber) {
            return new GeocodingResult(Result.Failed, Reason.InvalidHouseNumber, ProviderKey.KadasterAPI,
                providerType, `House number invalid - requested address cannot have empty house number`);
        }
 
        // Checking if the postCode is available
        if (!decodedAddress.postCode) {
            return new GeocodingResult(Result.Failed, Reason.InvalidPostcode, ProviderKey.KadasterAPI,
                providerType, `Missing Postcode`);
        }
 
        // Rejects addresses with empty streets and 4 character postcodes
        if ((!decodedAddress.street || decodedAddress.street === '') && decodedAddress.postCode.replace(' ', '').length === 4) {
            return new GeocodingResult(Result.Failed, Reason.InvalidStreet, ProviderKey.KadasterAPI,
                providerType, `Reject empty street with postcode length of 4`);
        }
 
        // Checking if the country code is correct
        if (decodedAddress.countryCode !== 'NL') {
            return new GeocodingResult(Result.Failed, Reason.NotMatchingParams, ProviderKey.KadasterAPI,
                providerType, `Kadaster only works for NL country codes`);
        }
 
        const searchAddress = this.buildKadasterAPIAddressSearchString(clientSettings, decodedAddress);
 
        let calls = 0;
 
        // Calling the parameter search api from Kadaster
        const requestOptions = {
            url: sourceEndPoint + 'search' + '?' + searchAddress,
            method: Method.GET,
            headers: [{
                name: 'Authorization',
                value: apiKey
            }]
        };
        logger.info(`Try request: ${requestOptions.url}`);
        const response = await makeRequest(requestOptions);
        calls += 1;
        if (response && response.status === 200) {
 
            let result = response.data.data;
 
            const cleanDecodedHouseNumber = decodedAddress.houseNumber.replace(/\D/g, '');
            const hasLettersInDecoded = /[a-zA-Z]/.test(decodedAddress.houseNumber);
 
            result = response.data.data.find((address: { huisnummer: string; huisletter: string; huisnummertoevoeging: string; }) => {
                const cleanHouseNumber = address.huisnummer.replace(/\D/g, '');
 
                // If the decoded address has letters, check for a return with letters
                if (hasLettersInDecoded) {
                    // Split the house number into numeric and letter parts
                    const match = decodedAddress.houseNumber?.match(/^(\d+)[-,]?([a-zA-Z]*)$/);
 
                    if (match) {
                        const numberPart = match[1];
                        const letterPart = match[2];
 
                        // Check if the number and letter match the corresponding fields
                        return numberPart === address.huisnummer && letterPart === (address.huisletter || address.huisnummertoevoeging);
                    }
                }
 
                // Else match the numbers only
                return cleanHouseNumber === cleanDecodedHouseNumber;
            }) as KadasterSearchResponse;
 
            const validation = this.validateKadasterResult(decodedAddress, result, clientSettings,
                postcodeCharRemoval);
            if (validation.valid) {
                calls += 1;
 
                // Calling the details api from Kadaster based off of an address returned from the parameter search
 
                const requestDetailedOptions = {
                    url: sourceEndPoint + 'details/' + result.id,
                    method: Method.GET,
                    headers: [{
                        name: 'Authorization',
                        value: apiKey
                    }]
                };
                logger.info(`Try request: ${requestDetailedOptions.url}`);
                let detailedResponse = await makeRequest(requestDetailedOptions);
 
                detailedResponse = detailedResponse.data.data;
 
                const { formattedAddress, detailedAddress } = this.getResultDetails(detailedResponse, countryFormat);
 
                if (detailedResponse.lat && detailedResponse.lon) {
                    const successfulResult = new GeocodingResult(
                        Result.Success,
                        Reason.SufficientAccuracy,
                        executionDefinition.provider.ref,
                        providerType,
                        `${validation.reason}`,
                        LocationType.Rooftop,
                        detailedResponse.lat,
                        detailedResponse.lon,
                        formattedAddress,
                        detailedAddress
                    );
                    successfulResult.score = defaultRooftopScore;
                    successfulResult.apiCalls = calls;
                    return successfulResult;
                }
            }
        }
 
        const failedResult = new GeocodingResult(
            Result.NotFound,
            Reason.NoResultFound,
            executionDefinition.provider.ref,
            providerType,
            `Api calls: ${calls}`
        );
        failedResult.apiCalls = calls;
        return failedResult;
    }
