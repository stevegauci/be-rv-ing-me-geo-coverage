    private initialRequestValidation(apiKey: string | undefined, sourceEndPoint: string | undefined, providerType: ProviderType,
        decodedAddress: DecodedAddress, executionDefinition: ProviderExecutionDefinition, streetMinCharLength: number | undefined) {
 
        // Checking if api details are missing from the provider settings
        if (apiKey === undefined || sourceEndPoint === undefined) {
            return new GeocodingResult(Result.Failed, Reason.MissingProviderOptions, ProviderKey.GoogleAPI,
                providerType, `One of 'apiKey' or 'sourceEndPoint was not defined in provider settings!`);
        }
 
        // Checking for a minimum street character length
        if (decodedAddress.address1 && streetMinCharLength) {
 
            // Clearing the country code before making the check to ensure that the street doesn't only contain country code
            const cleanAddress = decodedAddress.address1.split(' ').filter(word => word.toLowerCase()
                !== decodedAddress.countryCode?.toLowerCase()).join(' ');
 
            if (cleanAddress === '') {
                return new GeocodingResult(Result.Failed, Reason.NotMatchingParams, ProviderKey.GoogleAPI,
                    providerType, `Empty Street`);
            }
        }
 
        // Checking that the provider options exist for the requesting client.
        if (!executionDefinition.providerSettings.options) {
            return new GeocodingResult(Result.Failed, Reason.NotMatchingParams, ProviderKey.GoogleAPI, providerType);
        }
    }

        decodedAddress: DecodedAddress,

        const addressSearchStrings = [];
 
        const address = decodedAddress; // Change this!

            // If the above is valid then we have to check if the second UK postcode validation check is valid

            }

                newAddress1 = newAddress1.replace(/#/g, '');
 
                newAddress1 = newAddress1.split(' ').filter(word => word.toLowerCase() !== 'ar').join(' ');
                address.address1 = newAddress1;
            }
 
            addressParts.push(address.address1.trim());
        }
 
        if (address.address2 && searchAddressComponents.address2) {
            addressParts.push(address.address2.trim());
        }
 
        if (address.address3 && searchAddressComponents.address3) {
            addressParts.push(address.address3.trim());
        }
 
        if (address.address4 && searchAddressComponents.address4) {
            addressParts.push(address.address4.trim());
        }

            addressParts.push(address.postCode.trim());
        }

            addressParts.push(address.locality.trim());

        }
 
        if (addressParts.length > 0) {
            addressSearchStrings.push('address=' + addressParts.join(', '));
        }

    private determineLocationType(geocodingResult: GoogleGeocodingResult): LocationType {
        if (geocodingResult.geometry.location_type === 'ROOFTOP') {
            return LocationType.Rooftop;
        } else {
            const streetTypesArray = ['street_address', 'route', 'intersection', 'natural_feature', 'airport', 'park', 'point_of_interest'];
            const placeTypesArray = ['premise', 'subpremise'];
            if (geocodingResult.types.includes('postal_code')) {
                return LocationType.Postcode;
            } else if (geocodingResult.types.some((type) => placeTypesArray.includes(type))) {
                return LocationType.Rooftop;
            } else if (geocodingResult.types.some((type) => streetTypesArray.includes(type))) {
                return LocationType.Street;
            } else {
                return LocationType.Area;
            }
        }
    }
 
    private fuzzySearch(
        decodedAddress: DecodedAddress,
        googleGeocodingResult: GoogleGeocodingResult,
        fuzzySearchThreshold: number = 80
    ) {
 
        logger.info(`Running fuzzy search on google result`);
 
        let totalMatches: number = 0;
        let totalAddressParts: number = 0;
        const decodedAddressValues = [...new Set(Object.values(decodedAddress.originalAddress!))];
 
        const shortAddressComponents = googleGeocodingResult.address_components.map((x) => {
            return x.short_name;
        });
 
        const longAddressComponents = googleGeocodingResult.address_components.map((x) => {
            return x.long_name;
        });
 
        for (const addressValue of decodedAddressValues) {
 
            if (!addressValue || addressValue === '') {
                continue;
            }
 
            let splitAddress = this.cleanAddress(addressValue.split(' '));
 
            // If the address part is more than one word (to prevent execution on country code) then remove all 2 letter words
            // This eliminates words like 'de' which could lead to a higher score than 1.
            if (splitAddress.length > 1) {
                splitAddress = splitAddress.filter(word => !(word.length === 2 && /^[a-zA-Z]+$/.test(word)));
            }
 
            totalAddressParts = totalAddressParts + splitAddress.length;
 
            let matches: number = 0;
 
            for (let i = 0; i < googleGeocodingResult.address_components.length; i++) {
 
                const shortAddressComponent = shortAddressComponents[i];
                const longAddressComponent = longAddressComponents[i];
 
                let shortNameMatches = [];
                let longNameMatches = [];
 
                if (shortAddressComponent) {
                    const splitShortAddress = this.cleanAddress(shortAddressComponent.split(' '));
                    shortNameMatches = splitAddress.filter((x) => splitShortAddress.indexOf(x) !== -1);
                }
 
                if (longAddressComponent) {
                    const splitLongAddress = this.cleanAddress(longAddressComponent.split(' '));
                    longNameMatches = splitAddress.filter((x) => splitLongAddress.indexOf(x) !== -1);
                }
 
                if (shortNameMatches.length >= longNameMatches.length) {
                    matches = shortNameMatches.length;
                } else {
                    matches = longNameMatches.length;
                }
 
                totalMatches = totalMatches + matches;
            }
        }
 
        let score = Math.round((totalMatches / totalAddressParts) * 100) / 100;
 
        // Modifying the score in cases which the score is >1 leading to a final score > 100
        if (score > 1) { score = 1; }
 
        const valid = (score >= fuzzySearchThreshold) ? true : false;
        logger.info(`Fuzzy search score is: %${score}, and threshold is %${fuzzySearchThreshold}`);
        logger.info(`Fuzzy search on google result returned valid: ${valid}`);
 
        return {
            valid: valid,
            score: score
        };
    }
 
    private cleanAddress(addressPartsArray: string[]) {
 
        const abbreviations = [
            {
                abbreviation: /^rd$/,
                word: 'road'
            },
            {
                abbreviation: /^ln$/,
                word: 'lane'
            },
            {
                abbreviation: /^av$/,
                word: 'avenue'
            },
            {
                abbreviation: /^ave$/,
                word: 'avenue'
            },
            {
                abbreviation: /^st$/,
                word: 'street'
            },
            {
                abbreviation: /^str$/,
                word: 'street'
            },
            {
                abbreviation: /^cl$/,
                word: 'close'
            },
            {
                abbreviation: /^co$/,
                word: 'county'
            }
        ];
 
        const newAddressPartsArray = [];
 
        for (let addressPart of addressPartsArray) {
 
            addressPart = addressPart.trim();
            addressPart = addressPart.toLowerCase();
            addressPart = addressPart.replace(/[.*+?^=!:${}()|\[\]\/\\]/g, '');
            addressPart = addressPart.replace(/\s\s+/g, ' ');
 
            for (const abbreviationElement of abbreviations) {
                const { abbreviation, word } = abbreviationElement;
                addressPart = addressPart.replace(abbreviation, word);
            }
            newAddressPartsArray.push(addressPart);
        }
 
        let cleanAddress = newAddressPartsArray.join(' ');
        cleanAddress = cleanAddress.trim();
        cleanAddress = cleanAddress.toLowerCase();
 
        return cleanAddress.split(' ');
    }

    }
 
    protected validateGoogleResults(
        body: GeocodingResponse,

                    for (const postcodeSettings of clientSettings.addressValidation.postcodeSettings) {
 
                        // Retrieve the settings for the particular country code that requires these settings
                        if (postcodeSettings.countryCode === decodedAddress.countryCode || postcodeSettings.countryCode === 'ALL_CODES') {
 
                            // Setting the prefix check here since this is countryCode specific and not client
                            if (postcodeSettings.postcodePrefixCheck) {
                                postCodePrefix = postcodeSettings.postcodePrefixCheck;
                            }
                        }
                    }
                }
 
                if (postCodeResult.length > 0 && !postCodePrefix) {
 
                    //  Checking whether returned postcode matches requested postcode
                    if (postCodeResult[0].long_name !== decodedAddress.postCode
                        && postCodeResult[0].short_name !== decodedAddress.postCode) {
                        return {
                            index: i,
                            fuzzySearchScore: 0,
                            reason: Reason.NotMatchingParams
                        };
                    }
                }
 
                if (postCodePrefix) {
 
                    // Prefix check can be different from default for a client for specific postcode exceptions
                    if (clientSettings.postcodeExceptions &&
                        clientSettings.postcodeExceptions.postcodeList.includes(postCodeResult[0].long_name)) {
                        postCodePrefix = clientSettings.postcodeExceptions.postcodePrefixCheck;
                    }
 
                    const originalPostcode = decodedAddress.postCode?.substring(0, postCodePrefix);
                    const returnedPostcode = postCodeResult[0].long_name.substring(0, postCodePrefix);
 
                    // Checking whether the specified prefix of the requested postcode matches that of the response postcode
                    if (originalPostcode !== returnedPostcode) {
                        return {
                            index: i,
                            fuzzySearchScore: 0,
                            reason: Reason.NotMatchingParams
                        };
                    }
                }
            }
 
            const fuzzySearchResult = this.fuzzySearch(
                decodedAddress,
                googleGeocodingResult,
                fuzzySearchThreshold
            );
 
            if (!this.checkGoogleLocationBounds(googleGeocodingResult.geometry.bounds, clientSettings)) {
                fuzzySearchResult.valid = false;
                finalReason = Reason.NoValidLocation;
                resultIndexValue = i;
            }
 
            if (!clientSettings || !clientSettings.countryLimitations ||
                clientSettings.countryLimitations.length === 0 || !googleGeocodingResult.address_components) {
                fuzzySearchResult.valid = false;
                finalReason = Reason.NotMatchingParams;
                resultIndexValue = i;
            }
 
            if (googleGeocodingResult.address_components.some((x) =>
                x.types.includes('country'))) {

                    clientSettings?.countryLimitations?.includes(x.short_name))) {
                    fuzzySearchResult.valid = false;
                    finalReason = Reason.NotInRegionBounds;
                    resultIndexValue = i;
                }
            }
 
            // Checking whether the provider for the client has locationtype restriction
            if (allowedLocationTypes && allowedLocationTypes.length > 0 && googleGeocodingResult.geometry) {
                // Checking whether returned location type is acceptable
                if (!allowedLocationTypes.includes(googleGeocodingResult.geometry.location_type.toLowerCase())) {
                    return {
                        index: i,
                        fuzzySearchScore: fuzzySearchResult.score,
                        reason: Reason.NotAcceptedLocationType
                    };
                }
            }
 
            if (fuzzySearchResult.valid) {
                return {
                    index: i,
                    success: true,
                    fuzzySearchScore: fuzzySearchResult.score,
                    reason: Reason.SufficientAccuracy
                };
            }
        }
 
        // No result found.
        return {
            index: resultIndexValue,
            reason: finalReason
        };
    }
 
    protected checkGoogleLocationBounds(bounds: LatLngBounds, clientSettings: ClientGeocodingSettings): boolean {
        let valid = false;
 
        if (!bounds) {
            return true;
        }
 
        if (!clientSettings) {
            return true;
        }
 
        if (!clientSettings.locationBoundsThresholdMetres) {
            return true;
        }
 
        const neLat = bounds.northeast.lat;
        const neLong = bounds.northeast.lng;
        const swLat = bounds.southwest.lat;
        const swLong = bounds.southwest.lng;
 
        const dist = getDistanceFromLatLongInMetres(neLat, neLong, swLat, swLong);
        logger.info(`Distance ${dist}`);
        if (dist <= clientSettings.locationBoundsThresholdMetres) {
            valid = true;
        }
 
        return valid;
    }
 
    protected getDetailedAddress(validatedResult: GoogleGeocodingResult, countryFormat: string): DetailedAddress {
 
        const countryCode = validatedResult.address_components

            : '';

    async geocodeAddress(clientSettings: ClientGeocodingSettings,
        executionDefinition: ProviderExecutionDefinition, decodedAddress: DecodedAddress): Promise<GeocodingResult> {
 
        // Retrieving settings and address rules from stored geocoding settings
 
        // The google api key being used
        const apiKey = executionDefinition.providerSettings.options.get('apiKey');
 
        // The google api endpoint
        const sourceEndPoint = executionDefinition.providerSettings.options.get('sourceEndPoint');
 
        // Default timeout for the google response when provided
        const requestTimeout = executionDefinition.providerSettings.options.get('requestTimeout') as number | undefined;
 
        // Default fuzzy search threshold for the google response when provided
        const fuzzySearchThreshold = executionDefinition.providerSettings.options.get('fuzzySearchThreshold') as number | undefined;
 
        // Location types allowed for the responses for the requesting client
        const allowedLocationTypes = executionDefinition.providerSettings.options.get('allowedLocationTypes') as string[] | undefined;
 
        // Boolean value stating whether to match on postcode
        const postcodeMatch = executionDefinition.providerSettings.options.get('postcodeMatch') as boolean | undefined;
 
        // Location types rejected for the responses for the requesting client
        const rejectLocationTypes = executionDefinition.providerSettings.options.get('rejectLocationTypes') || [] as LocationType[];
 
        // The provider type for this particular provider - in this case would be 3rdparty
        const providerType = executionDefinition.providerSettings.providerType;
 
        const providerCountryCodeFormat = executionDefinition.providerSettings.options.get('countryCodeFormat') || countryCodeFormat.Alpha2;
 
        const countryFormat = clientSettings.countryCodeFormat || countryCodeFormat.Alpha2;
 
        // List of characters to be removed from postcodes for specific clients
        const postcodeCharRemoval = executionDefinition.providerSettings.options.get('postcodeCharRemoval') as {
            charactersToRemove: [],
            removeAllLetters: boolean
        } | undefined;
 
        // The minimum number of characters allowed within a street field
        const streetMinCharLength = clientSettings.streetMinCharLength as number | undefined;
 
        // Handling all initial request validation to determine whether the address should even pass through this provider
        const requestValidation = this.initialRequestValidation(apiKey, sourceEndPoint, providerType,
            decodedAddress, executionDefinition, streetMinCharLength);
 
        // If any validation failures are given then return the validation failure
        if (requestValidation) { return requestValidation; }
 
        let apiCalls = 0;
        let intermediateResult;
        let addressesArray: string[] = [];
        const fuzzySearchScores: number[] = [0];
        const googleResults = [];
 
        if (decodedAddress.address) {
            addressesArray.push(`address=${decodedAddress.address}`);
        } else {
            addressesArray = this.buildGoogleAddressSearches(clientSettings, decodedAddress, providerCountryCodeFormat);
        }
 
        for (const address of addressesArray) {
 
            // Building the google api request
            const requestOptions = {
                url: `${sourceEndPoint}?${address}&key=${apiKey}`,
                method: Method.GET,
                timeout: requestTimeout
            };
 
            logger.info(`Try request: ${address}`);
            const urlLength = encodeURI(requestOptions.url).length;
 
            if (urlLength > this.maxUrlChars) {
                logger.warn(`Request exceeds maximum length! (${this.maxUrlChars})`);
            }
 
            // This tracks the number of api calls within the logs for billing purposes
            apiCalls += 1;
            let response;
 
            try {
                // Carrying out the google api request
                response = await makeRequest(requestOptions);
            } catch (err) {
                logger.info(`Re-trying request: ${address}`);
 
                try {
                    // If request fails then re-attempt
                    response = await makeRequest(requestOptions);
                    logger.info('Re-trying request worked!');
                } catch (err) {
                    logger.error(`Re-trying request failed: ${err}`);
                    return new GeocodingResult(Result.Rejected, Reason.ServiceUnreachable, ProviderKey.GoogleAPI, providerType);
                }
            }
 
            if (!response) {
                logger.info(`No response from Google`);
                return new GeocodingResult(Result.Rejected, Reason.ServiceUnreachable, ProviderKey.GoogleAPI, providerType);
            }
 
            logger.info(`Got response from Google, status code: ${response.status}`);
 
            if (response.status !== 200) {
                // Unreachable or malicious
                return new GeocodingResult(Result.Rejected, Reason.ServiceUnreachable, ProviderKey.GoogleAPI, providerType);
            }
 
            const result = response.data as GeocodingResponse;
            const validation = this.validateGoogleResults(result, clientSettings, fuzzySearchThreshold,
                decodedAddress, allowedLocationTypes, postcodeMatch);
 
            const fuzzySearchScore = validation.fuzzySearchScore ? validation.fuzzySearchScore : 0;
            fuzzySearchScores.push(fuzzySearchScore);
 
            const addressComponents = result.results && result.results[0] &&
                result.results[0].address_components ? result.results[0].address_components : {};
 
            const googleResult = {
                score: fuzzySearchScore,
                addressComponents: addressComponents
            };
 
            googleResults.push(googleResult);
 
            let addressLocationType;
 
            if (validation.success) {
 
                addressLocationType = this.determineLocationType(result.results[validation.index]);
            }
 
            logger.info(`Google validation done`);
 
            const validatedResult = result.results[validation.index] ?? undefined;
            const longitude = validatedResult?.geometry?.location?.lng;
            const latitude = validatedResult?.geometry?.location?.lat;
            const detailedAddress = validatedResult ? this.getDetailedAddress(validatedResult, countryFormat) : undefined;
            let formattedAddress = validatedResult?.formatted_address;
 
            if (addressLocationType && !rejectLocationTypes.includes(addressLocationType) && detailedAddress) {
 
                let decodedPostcode = decodedAddress.postCode;
                let detailedPostcode = detailedAddress.postcode || '';
 
                // Remove characters from the postcodes
                if (postcodeCharRemoval) {
 
                    if (postcodeCharRemoval.removeAllLetters) {
                        decodedPostcode = detailedAddress.postcode ? detailedAddress.postcode.replace(/\D/g, '') : '';
                        detailedPostcode = decodedAddress.postCode ? decodedAddress.postCode.replace(/\D/g, '') : '';
                    } else {
                        for (const pcChar of postcodeCharRemoval.charactersToRemove) {
 
                            decodedPostcode = detailedAddress.postcode ? detailedAddress.postcode.replace(pcChar, '') : '';
                            detailedPostcode = decodedAddress.postCode ? decodedAddress.postCode.replace(pcChar, '') : '';
                        }
                    }
 
                    formattedAddress = detailedAddress.postcode ? formattedAddress.replace(detailedAddress.postcode, detailedPostcode)
                        : formattedAddress;
 
                    detailedAddress.postcode = detailedPostcode;
                }
 
                let trustedResult;
                let reason = Reason.SufficientAccuracy;
 
                if (executionDefinition.providerSettings.applyTrustedResult &&
                    detailedPostcode &&
                    detailedPostcode === decodedPostcode) {
                    logger.info(`This result is trusted based on postcode match!`);
                    trustedResult = true;
                    reason = Reason.TrustedResult;
                }
 
                const successfulResult = new GeocodingResult(
                    Result.Success, reason,
                    ProviderKey.GoogleAPI,
                    providerType,
                    ``,
                    addressLocationType,
                    latitude, longitude,
                    formattedAddress,
                    detailedAddress,
                    undefined, undefined,
                    trustedResult
                );
                successfulResult.apiCalls = apiCalls;
                successfulResult.score = Math.max(...fuzzySearchScores);
                successfulResult.highestScoringAddress = googleResults.find(x => x.score === successfulResult.score);
 
                return successfulResult;
            } else {
                intermediateResult = new GeocodingResult(
                    Result.Rejected, validation.reason,
                    ProviderKey.GoogleAPI,
                    providerType,
                    ``,
                    addressLocationType,
                    latitude, longitude,
                    formattedAddress,
                    detailedAddress,
                    undefined, undefined,
                    false
                );
                intermediateResult.apiCalls = apiCalls;
                intermediateResult.score = Math.max(...fuzzySearchScores);
                intermediateResult.highestScoringAddress = googleResult;
            }
        }
 
        if (intermediateResult) {
            return intermediateResult;
        }
 
        const failedResult
            = new GeocodingResult(Result.Failed, Reason.NoResultFound, ProviderKey.GoogleAPI, providerType, `Api calls: ${apiCalls}`);
        failedResult.apiCalls = apiCalls;
        failedResult.score = Math.max(...fuzzySearchScores);
        failedResult.highestScoringAddress = googleResults.find(x => x.score === failedResult.score);
        return failedResult;
    }
