import { Service } from 'typedi';
import { HttpError } from '@loqusgroup/sls-openfleet-common';
import { ProviderProfile } from '../../database/data-models';
import { IndexSearchAxios } from '../provider-execution/index-search-axios.service';
 
@Service()
export class ProviderProfileService {
 
    /*
        Creating a provider profile
 
        @param providerProfile : The provider profile object to be created
        @returns : The status of the creation process
 
    */
    async createProviderProfile(providerProfile: ProviderProfile) {
 
        const providerProfileRepository = process.env.PROVIDERPROFILES;
        const createUrl = process.env.IX_CREATE || '';
 
        const indexSearchService = new IndexSearchAxios();
 
        // Passing parameters to a generic service that will carry out the call to the respective index
        const result = await indexSearchService.executeQuery(1, providerProfile, 'POST', providerProfileRepository + createUrl);
 
        if (!result) {
            return;
        }
 
        return result;
    }
 
    /*
        Retrieving the provider profiles
 
        @param id : The id matching the requested provider profile
        @returns : The provider profile object, or else a null value if not found.
 
    */
    async getProviderProfile(profileKey: string | null, limit: number = 1) {
 
        let providerProfilesQuery = {};
 
        // Query to search for the provider profile by id
        if (profileKey !== null) {
            providerProfilesQuery = {
                'match': {
                    'profileKey': {
                        'query': profileKey
                    }
                }
            };
        }
 
        // Getting the index endpoint to be used for this provider
        const indexUrl = process.env.PROVIDERPROFILES || '';
 
        const searchUrl = process.env.IX_SEARCH || '';
 
        if (!indexUrl || !searchUrl) {
            throw new HttpError.InternalServerError(`Invalid provider profiles index URL ${indexUrl}`);
        }
 
        // Assigning the method for the request for this provider
        const method = 'GET';
 
        const indexSearchService = new IndexSearchAxios();
 
        // Passing parameters to a generic service that will carry out the call to the respective index
        let dbResults = await indexSearchService.executeQuery(limit, providerProfilesQuery, method, indexUrl + searchUrl);
        dbResults = dbResults.hits.hits;
 
        // Iterate through all documents to remove apiKey from providerSettings
        for (const doc of dbResults) {
            if (doc._source && Array.isArray(doc._source.providerSettings)) {
                for (const provider of doc._source.providerSettings) {
                    if (provider.options && provider.options.apiKey) {
                        delete provider.options.apiKey;
                    }
                }
            }
        }
 
        const hasSingleResult = Array.isArray(dbResults) && dbResults.length > 0 && limit === 1;
 
        if (hasSingleResult) {
            dbResults[0]._source.id = dbResults[0]._id;
            return dbResults[0]._source;
        }
 
        return dbResults;
    }
 
    async deleteProviderProfile(id: string) {
 
        const indexSearchService = new IndexSearchAxios();
        const cacheRepository = process.env.PROVIDERPROFILES || '';
        const deleteUrl = process.env.IX_DELETE;
 
        if (id) {
            try {
                const result = await indexSearchService.executeQuery(1, {}, 'DELETE', cacheRepository + deleteUrl + id);
                return result;
            } catch (error) {
                throw new HttpError.NotFound();
            }
 
        } else {
            throw new HttpError.NotFound();
        }
    }
 
    async updateProviderProfile(profileKey: string, providerProfile: ProviderProfile) {
 
        const currentProviderProfile = await this.getProviderProfile(profileKey, 1);
 
        const isEmptyArray = Array.isArray(currentProviderProfile) && currentProviderProfile.length === 0;
 
        if (!currentProviderProfile || isEmptyArray) {
            throw new HttpError.NotFound();
        }
 
        const providerProfileRepository = process.env.PROVIDERPROFILES || '';
        const updateUrl = process.env.IX_UPDATE || '';
 
        const indexSearchService = new IndexSearchAxios();
 
        const scriptStatements: string[] = [];
        const params: Record<string, any> = { updates: [] };
 
        params.currentTime = new Date().toISOString();
 
        scriptStatements.push(`
            if (ctx._source.changeLog == null) {
                ctx._source.changeLog = [];
            }
 
            def previousState = new HashMap();
            for (entry in ctx._source.entrySet()) {
                if (entry.getKey() != 'changeLog') {
                    previousState.put(entry.getKey(), entry.getValue());
                }
            }
            previousState.updatedAt = params.currentTime;
            ctx._source.changeLog.add(previousState);
        `);
        
        if (providerProfile.providerSettings && Array.isArray(providerProfile.providerSettings)) {
            providerProfile.providerSettings.forEach((updateObject, index) => {
                params.updates.push(updateObject);
 
                scriptStatements.push(`
                for (int i = 0; i < ctx._source.providerSettings.length; i++) {
                    if (ctx._source.providerSettings[i].providerKey != null &&
                        ctx._source.providerSettings[i].providerKey == params.updates[${index}].providerKey) {
                        
                        for (entry in params.updates[${index}].entrySet()) {
                            if (entry.getKey() != 'options') {
                                ctx._source.providerSettings[i][entry.getKey()] = entry.getValue();
                            } else if (entry.getKey() == 'options') {
                                for (opt in entry.getValue().entrySet()) {
                                    ctx._source.providerSettings[i].options[opt.getKey()] = opt.getValue();
                                }
                            }
                        }
                    }
                }
            `);
            });
        }
 
        for (const field in providerProfile) {
            if (field !== 'providerSettings') {
                scriptStatements.push('ctx._source.' + field + ' = params.' + field + ';');
                params[field] = (providerProfile as any)[field];
            }
        }
 
 
        const query = {
            'query': {
                'term': {
                    'profileKey': profileKey
                }
            },
            'script': {
                source: scriptStatements.join('\n'),
                'lang': 'painless',
                'params': params
            }
        };
 
        // Passing parameters to a generic service that will carry out the call to the respective index
        const dbResults = await indexSearchService.executeQuery(1, query, 'POST', providerProfileRepository + updateUrl);
 
        if (!dbResults) {
            return;
        }
 
        return dbResults;
    }
 
    async list(limit: number | undefined)
        : Promise<ProviderProfile[]> {
        return await this.getProviderProfile(null, limit ? limit : 100);
    }
 
}
