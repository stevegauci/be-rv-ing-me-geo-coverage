"use strict";
 
const path = require("path");
 
// Services
const ReportRequestsService = require(path.resolve("./services/report-requests.service"));
 
/* ---------------------------------------- REPORTS REQUEST CONTROLLER ---------------------------------------- */
 
module.exports = function (app) {
 
    let reportRequestsService;
 
    let methods = {};
 
    methods.context = async function (req, res, next) {
 
        reportRequestsService = new ReportRequestsService(app, req.client, req);
 
        return next();
    };
    
    /**
     * List report request API
     * 
     * @example {{of-kpi-api}}/report-requests?reportType=geocoding
     * @method GET
     * 
     * @see https://lqs-backend-team.postman.co/workspace/Lqs-Backend-Devs~7ff6dfeb-3c7c-4985-addd-9ab8b5cc57d7/request/19398392-eed5c669-937f-41dd-9b32-08bafa66156a?action=share&source=copy-link&creator=19398392&active-environment=634d3b4a-c53f-4ba0-9155-e62228f3c9de
     * @see services/report-requests.service.js list()
     * 
     * @author @nichmica7
     * 
     * @param {object} req - The request object, containing query parameters and request headers.
     * {
     *   query: {
     *     reportType: String, // Optional report type to filter the requests
     * }
     * @param {object} res - The response object used to send back the HTTP response.
    */
    methods.list = async function (req, res) {
 
        const { err, result } = await reportRequestsService.list(req);
 
        if (err) {
            return res.status(400).send({
                message: err.message
            });
        }
 
        return res.send(result);
 
    };
    
    /**
     * Create report request API
     * 
     * @example {{of-kpi-api}}/report-requests
     * @method POST
     * 
     * @see https://lqs-backend-team.postman.co/workspace/Lqs-Backend-Devs~7ff6dfeb-3c7c-4985-addd-9ab8b5cc57d7/request/19398392-7a63edf9-41d4-4abd-b7cd-d426b12957ae?action=share&source=copy-link&creator=19398392&active-environment=634d3b4a-c53f-4ba0-9155-e62228f3c9de
     * @see services/report-requests.service.js create()
     * 
     * @author @nichmica7
     * 
     * @param {Object} req - The request object containing all necessary data in the body. Expected structure:
     * {
     *   body: {
     *     type: String, // Type of the report to be generated
     *     filter: Object, // Filters to be applied when generating the report
     *     outputType: String, // Format of the output (e.g., 'excel', 'json', 'csv')
     *     additionalFields: Object, // Optional additional fields for report generation
     *   }
     * }
     * 
     * @param {object} res - The response object used to send back the HTTP response.
     * 
     * @returns 
     */
    methods.create = async function (req, res) {
 
        const { err, result } = await reportRequestsService.create(req);
 
        if (err) {
            return res.status(400).send({
                message: err.message
            });
        }
 
        return res.send(result);
    };
 
    /**
     * Downloads a report file from cloud storage.
     * 
     * @example {{of-kpi-api}}/report-requests/download?reportId=bc1b7ee0-0b81-11ef-b93a-a73d3689bc8d
     * @method GET
     * 
     * @see https://lqs-backend-team.postman.co/workspace/Lqs-Backend-Devs~7ff6dfeb-3c7c-4985-addd-9ab8b5cc57d7/request/19398392-9a19cae7-e9d3-4d57-b190-c5533b0e3727?action=share&source=copy-link&creator=19398392&active-environment=634d3b4a-c53f-4ba0-9155-e62228f3c9de
     * @see services/report-requests.service.js download()
     * 
     * @author @nichmica7
     * 
     * @param {Object} req - The request object.
     * @param {Object} res - The response object.
     */
    methods.download = async function (req, res) {
 
        const { err, result } = await reportRequestsService.download(req);
 
        if (err) {
            return res.status(400).send({
                message: err.message
            });
        }
 
        return res.send(result);
    };
 
    return methods;
};
