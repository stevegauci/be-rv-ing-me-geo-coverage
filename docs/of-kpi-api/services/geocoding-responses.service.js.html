"use strict";
 
const path = require("path");
 
const geocodingResponseElasticOpenSearch = require(path.resolve("./elasticopensearch/queries/geocoding-responses.queries"));
 
const openfleetCore = require("openfleet-core");
 
const redisUtils = openfleetCore.lib.redisUtils;
 
const OpenSearchLogic = openfleetCore.logic.Opensearch;
 
const ClientSettingModel = openfleetCore.models.ClientSetting.model;
 
module.exports = class GeocodingResponsesService {
 
    constructor(app, client, req) {
 
        this.app = app;
        this.req = req;
        this.client = client;
        this.db = app.get("db");
        this.l10n = app.get("l10n");
        this.logger = app.get("logger");
        this.config = app.get("config");
        this.cacheService = app.get("cacheService");
        this.clientSetting = ClientSettingModel(this.db, client);
 
        this.logger.debug("creating geocoding response service");
    }
 
    /**
     * Retrieves a list of geocoding responses within a specific timeframe for a specific client.
     * 
     * @author @dylanbulteel
     *
     * @param {string} req.query.dateFrom - the dateFrom value is a date
     * @param {string} req.query.dateTo - the dateTo value is a date
     *
     * @returns {Promise<Object>} Returns a promise that resolves to an object containing either 'result' or 'err'.
     * If 'result' is present, it contains the geocoding response. If 'err' is present, it contains error details.
     */
    async get(options) {
 
        try {
 
            const clientSettings = await redisUtils.loadDataFromCache(this, ClientSettingModel);
 
            if (!clientSettings) {
                throw new Error(this.l10n.t(this.req.headers["accept-language"], "beClientSettingsNotFound", "Client settings not found"));
            }
 
            if (!clientSettings.geocodingSettings?.failureReasonBuckets) {
                throw new Error(this.l10n.t(this.req.headers["accept-language"], "beNoGeocodingSettingsFoundForClientID", "No Geocoding settings found for client ID: {1}", [this.client._id]));
            }
 
            const openSearchLogic = OpenSearchLogic(this.app, this.client);
 
            const { req, indexName } = options;
 
            const failureReasonBuckets = clientSettings.geocodingSettings?.failureReasonBuckets;
 
            const elasticSearchOptions = {
                indexName: indexName.toUpperCase(),
                finalQuery: geocodingResponseElasticOpenSearch.get({ req, limit: 0, failureReasonBuckets }),
                clientId: req.client._id,
                req
            };
 
            const { err, result } = await openSearchLogic.read(elasticSearchOptions);
 
            if (err) {
                this.logger.error(err.message);
                return { err };
            }
 
            if (!result || !result.data) {
                this.logger.error("Index not created!");
                return { result: {} };
            }
 
            // Reading the aggregation results
            const aggregationResults = result.data.aggregations;
 
            const successfulResults = aggregationResults.successfulResults;
            let totalSuccessfulResults = successfulResults.doc_count;
 
            // Building the summary object of the entire set of documents
            const summary = {
                "dateFrom": req.query.dateFrom,
                "dateTo": req.query.dateTo,
                "noOfJobs": aggregationResults.noOfJobs.value,
                "noOfGeocodingRequests": aggregationResults.noOfGeocodingRequests.value,
                "provider.local": aggregationResults.provider_local.doc_count,
                "provider.3rdParty": aggregationResults.provider_3rdparty.doc_count,
                "successfulResults": totalSuccessfulResults,
                "failedResults": aggregationResults.failedResults.doc_count,
                "rooftopCount": successfulResults.rooftopCount.doc_count,
                "rooftopPercentage": await this.calculatePercentage(successfulResults.rooftopCount.doc_count, totalSuccessfulResults),
                "rooftopAvgScore": Math.min(successfulResults.rooftopCount?.avgRooftopScore?.value ?? 0, 1),
                "streetCount": successfulResults.streetCount.doc_count,
                "streetPercentage": await this.calculatePercentage(successfulResults.streetCount.doc_count, totalSuccessfulResults),
                "streetAvgScore": Math.min(successfulResults.streetCount?.avgStreetScore?.value ?? 0, 1),
                "streetAccessPointCount": successfulResults.streetAccessPointCount.doc_count,
                "streetAccessPointPercentage": await this.calculatePercentage(successfulResults.streetAccessPointCount.doc_count, totalSuccessfulResults),
                "streetAccessPointAvgScore": Math.min(successfulResults.streetAccessPointCount?.avgStreetAccessPointScore?.value ?? 0, 1),
                "postcodeCount": successfulResults.postcodeCount.doc_count,
                "postcodePercentage": await this.calculatePercentage(successfulResults.postcodeCount.doc_count, totalSuccessfulResults),
                "postcodeAvgScore": Math.min(successfulResults.postcodeCount?.avgPostcodeScore?.value ?? 0, 1),
                "postcodeSectorCount": successfulResults.postcodeSectorCount.doc_count,
                "postcodeSectorPercentage": await this.calculatePercentage(successfulResults.postcodeSectorCount.doc_count, totalSuccessfulResults),
                "postcodeSectorAvgScore": Math.min(successfulResults.postcodeSectorCount?.avgPostcodeSectorScore?.value ?? 0, 1),
                "areaCount": successfulResults.areaCount.doc_count,
                "areaPercentage": await this.calculatePercentage(successfulResults.areaCount.doc_count, totalSuccessfulResults),
                "areaAvgScore": Math.min(successfulResults.areaCount?.avgAreaScore?.value ?? 0, 1),
                "successfulResultsPercentage": await this.calculatePercentage(totalSuccessfulResults, totalSuccessfulResults + aggregationResults.failedResults.doc_count),
                "serviceIssueCount": aggregationResults.failedResults.serviceIssueCount.doc_count,
                "noResultCount": aggregationResults.failedResults.noResultCount.doc_count,
                "rejectedParametersCount": aggregationResults.failedResults.rejectedParametersCount.doc_count,
                "insufficientAccuracyCount": aggregationResults.failedResults.insufficientAccuracyCount.doc_count,
            };
 
            let breakdown = [];
 
            // Building the breakdown object containing information on a depot level
            for (const depot of aggregationResults.depotRef_stats.buckets) {
 
                const depot_successfulResults = depot.successfulResults;
                let depot_totalSuccessfulResults = depot_successfulResults.doc_count;
 
                const depotBreakdown = {
                    "depotRef": depot.key,
                    "depotName": depot.depotName.buckets[0].key,
                    "noOfJobs": depot.noOfJobs.value,
                    "noOfGeocodingRequests": depot.noOfGeocodingRequests.value,
                    "provider.local": depot.provider_local.doc_count,
                    "provider.3rdParty": depot.provider_3rdparty.doc_count,
                    "successfulResults": depot_totalSuccessfulResults,
                    "failedResults": depot.failedResults.doc_count,
                    "rooftopCount": depot_successfulResults.rooftopCount.doc_count,
                    "rooftopPercentage": await this.calculatePercentage(depot_successfulResults.rooftopCount.doc_count, depot_totalSuccessfulResults),
                    "rooftopAvgScore": Math.min(depot_successfulResults.rooftopCount?.avgRooftopScore?.value ?? 0, 1),
                    "streetCount": depot_successfulResults.streetCount.doc_count,
                    "streetPercentage": await this.calculatePercentage(depot_successfulResults.streetCount.doc_count, depot_totalSuccessfulResults),
                    "streetAvgScore": Math.min(depot_successfulResults.streetCount?.avgStreetScore?.value ?? 0, 1),
                    "streetAccessPointCount": depot_successfulResults.streetAccessPointCount.doc_count,
                    "streetAccessPointPercentage": await this.calculatePercentage(depot_successfulResults.streetAccessPointCount.doc_count, depot_totalSuccessfulResults),
                    "streetAccessPointAvgScore": Math.min(depot_successfulResults.streetAccessPointCount?.avgStreetAccessPointScore?.value ?? 0, 1),
                    "postcodeCount": depot_successfulResults.postcodeCount.doc_count,
                    "postcodePercentage": await this.calculatePercentage(depot_successfulResults.postcodeCount.doc_count, depot_totalSuccessfulResults),
                    "postcodeAvgScore": Math.min(depot_successfulResults.postcodeCount?.avgPostcodeScore?.value ?? 0, 1),
                    "postcodeSectorCount": depot_successfulResults.postcodeSectorCount.doc_count,
                    "postcodeSectorPercentage": await this.calculatePercentage(depot_successfulResults.postcodeSectorCount.doc_count, depot_totalSuccessfulResults),
                    "postcodeSectorAvgScore": Math.min(depot_successfulResults.postcodeSectorCount?.avgPostcodeSectorScore?.value ?? 0, 1),
                    "areaCount": depot_successfulResults.areaCount.doc_count,
                    "areaPercentage": await this.calculatePercentage(depot_successfulResults.areaCount.doc_count, depot_totalSuccessfulResults),
                    "areaAvgScore": Math.min(depot_successfulResults.areaCount?.avgAreaScore?.value ?? 0, 1),
                    "successfulResultsPercentage": await this.calculatePercentage(depot_totalSuccessfulResults, depot_totalSuccessfulResults + depot.failedResults.doc_count),
                    "serviceIssueCount": depot.failedResults.serviceIssueCount.doc_count,
                    "noResultCount": depot.failedResults.noResultCount.doc_count,
                    "rejectedParametersCount": depot.failedResults.rejectedParametersCount.doc_count,
                    "insufficientAccuracyCount": depot.failedResults.insufficientAccuracyCount.doc_count,
                };
 
                breakdown.push(depotBreakdown);
            }
 
            return { result: { summary: summary, breakdown: breakdown } };
 
        } catch (err) {
            this.logger.error(`Error geocoding response .get() - ${err.message}`);
            return { err };
        }
    }
 
    // Function to calculate count percentages
    async calculatePercentage(count, total) {
 
        if (total === 0) {
            return 0;
        }
        const calculation = parseFloat(((count / total) * 100).toFixed(1));
        return calculation;
    }
};
