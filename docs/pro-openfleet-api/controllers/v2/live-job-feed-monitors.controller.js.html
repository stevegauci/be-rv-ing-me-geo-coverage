"use strict";
 
const openfleetCore = require("openfleet-core");
 
const CommonHandler = openfleetCore.handlers.Common;
 
const LiveJobFeedMonitorModel = openfleetCore.models.LiveJobFeedMonitor.model;
const ClientSettingsModel = openfleetCore.models.ClientSetting.model;
 
module.exports = function (app) {
 
    const db = app.get("db");
    const cacheService = app.get("cacheService");
    const l10n = app.get("l10n");
 
    //Models
    let clientSettings;
    let liveJobFeedMonitorHandler;
 
    const methods = {};
 
    methods.context = function (req, res, next) {
 
        clientSettings = ClientSettingsModel(db, req.client);
        liveJobFeedMonitorHandler = CommonHandler(app, req.client, LiveJobFeedMonitorModel);
 
        return next();
    };
 
    methods.list = function (req, res) {
 
        liveJobFeedMonitorHandler.list({ filter: {} }, function (err, liveJobFeedMonitors) {
            if (err) {
                return res.status(400).send({
                    message: err.message
                });
            }  
 
            if(!liveJobFeedMonitors){
                return res.status(400).send({
                    message: l10n.t(req.headers["accept-language"], "beNoLiveJobFeedMonitorsFound", "No Live Job Feed Monitors Found")
                });
            }
 
            clientSettings.fromCache(cacheService, req.client._id, function (err, clientSettings) {
                if (err) {
                    return res.status(400).send({
                        message: err.message
                    });
                }
                let thresholds;
 
                if (clientSettings) {
                    thresholds = clientSettings.liveJobFeedThresholds;
                }
 
                return res.send({ liveJobFeedMonitors: liveJobFeedMonitors, liveJobFeedThreshold: thresholds });
            });
        });
    };
 
    return methods;
};
