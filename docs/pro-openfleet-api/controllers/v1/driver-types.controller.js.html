    function isUsed(filter, req, callback) {
 
        const isUsedAgg = driverTypeAgg.listResourcesThatMatchingDriverType({ driverTypeId: filter._id });
 
        clientSettingsCommonHandler.aggregate({ filter: isUsedAgg }, function (err, isUsed) {
 
            if (err) {
                return callback(err, null);
            }
 
            const driverCheck = isUsed && isUsed.length > 0 ? isUsed[0] : null;
            if (!driverCheck) {
                return callback(null, true, l10n.t(req.headers["accept-language"], "lblDataNotLoaded", "Data not loaded"));
            }
 
            if (driverCheck.drivers.length > 0) {
                return callback(null, true, l10n.t(req.headers["accept-language"], "beDeleteProhibitedDriverTypeLinkedWithADriver", "Delete prohibited: Driver type is linked with a driver"));
            }
 
            if (driverCheck.resourceprofiles.length > 0) {
                return callback(null, true, l10n.t(req.headers["accept-language"], "beDeleteProhibitedDriverTypeLinkedWithResourceProfile", "Delete prohibited: Driver type is linked with a resource profiles"));
            }
 
            if (driverCheck.resourcecalendars.length > 0) {
                return callback(null, true, l10n.t(req.headers["accept-language"], "beDeleteProhibitedDriverTypeLinkedWithResourceCalendar", "Delete prohibited: Driver type is linked with a resource calendar"));
            }
 
            if (driverCheck.resourcecalendartemplates.length > 0) {
                return callback(null, true, l10n.t(req.headers["accept-language"], "beDeleteProhibitedDriverTypeLinkedWithResourceCalendarTemplate", "Delete prohibited: Driver type is linked with a resource calendar template"));
            }
 
            if (driverCheck.dispatchmanifests.length > 0) {
                return callback(null, true, l10n.t(req.headers["accept-language"], "beDeleteProhibitedDriverTypeLinkedWithDispatchManifest", "Delete prohibited: Driver type is linked with a dispatch manifest"));
            }
 
            return callback(null, false);
        });
    }
