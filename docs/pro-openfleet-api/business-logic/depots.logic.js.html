"use strict";
 
const dayjs = require("dayjs");
const mongoose = require("mongoose");
const { promisify } = require("util");
const openfleetCore = require("openfleet-core");
 
// consts
const scenarioConst = openfleetCore.consts.scenarioConst;
 
// models
const SettingsModel = openfleetCore.models.Setting.model;
const HolidaysModel = openfleetCore.models.Holiday.model;
 
// handlers
const AsyncCommonHandler = openfleetCore.handlers.AsyncCommon;
 
/* ---------------------------------------- DEPOTS LOGIC ---------------------------------------- */
 
module.exports = function (app, client) {
 
    const db = app.get("db");
    const l10n = app.get("l10n");
    const cacheService = app.get("cacheService");
 
    const DepotSettings = SettingsModel(db, client);
 
    const methods = {};
 
    /**
     * @description Retrieves the operational status for a given depot on a specified day of week or date.
     * @author Steve Gauci
     * 
     * @param {Object} options - Options object containing depotId, date, dayOfWeek, and reqLanguage.
     * @param {string} options.depotId - The ID of the depot.
     * @param {string} [options.date] - The specific date to check in YYYY-MM-DD format. Optional.
     * @param {number} [options.dayOfWeek] - The day of the week to check (0 for Sunday, 1 for Monday, etc.). Optional.
     * @param {string} options.reqLanguage - The language code for the request.
     * 
     * @returns {Object} - The operational status of the depot or an error message.
     * 
     * @example
     * // Example usage:
     * // getOperationalStatus({ depotId: '7447', dayOfWeek: 3, reqLanguage: 'en' })
     * // getOperationalStatus({ depotId: '7447', date: '2024-05-22', reqLanguage: 'en' })
     */
    methods.getOperationalStatus = async function (options) {
 
        const { depotId, date, dayOfWeek, reqLanguage } = options;
 
        try {
 
            const operationalStatus = {
                operationalDay: {},
            };
 
            const extractedDayOfWeek = dayOfWeek && !isNaN(parseInt(dayOfWeek)) ? parseInt(dayOfWeek) : date && dayjs(date).isValid() ? dayjs(date).day() : null;
 
            if ([null, undefined].includes(extractedDayOfWeek)) {
                throw new Error(l10n.t(reqLanguage, "beUnableToObtainDayOfWeek", "Unable To Obtain Day of Week"));
            }
 
            DepotSettings.fromCache = promisify(DepotSettings.fromCache);
            const depotSettings = await DepotSettings.fromCache(cacheService, client._id, depotId);
 
            if (!depotSettings) {
                throw new Error(l10n.t(reqLanguage, "beUnableToObtainDepotSettings", "Unable To Obtain Depot Settings"));
            }
 
            if (!depotSettings.operationalDays?.length) {
                throw new Error(l10n.t(reqLanguage, "beUnableToObtainOperationalDaysDepotSettings", "Unable to Obtain Operational Days from Depot Settings"));
            }
 
            const operationalDay = depotSettings.operationalDays.find(operationalDay => operationalDay.dayOfWeek == extractedDayOfWeek);
 
            if (operationalDay) {
                operationalStatus.operationalDay = operationalDay;
                delete operationalStatus.operationalDay.dayOfWeek;
            }
 
            if (date) {
                const parsedDate = dayjs.utc(date);
 
                const holidaysHandler = await AsyncCommonHandler(app, client, HolidaysModel);
                const readOptions = {
                    filter: {
                        $and: [
                            {
                                $or: [
                                    {
                                        depots: {
                                            $in: [mongoose.Types.ObjectId(depotId)]
                                        }
                                    },
                                    {
                                        allDepots: true
                                    }
                                ]
                            },
                            {
                                $or: [
                                    {
                                        date: parsedDate
                                    },
                                    {
                                        $and: [
                                            { $expr: { $eq: [{ $month: "$date" }, parsedDate.month() + 1] } },
                                            { $expr: { $eq: [{ $dayOfMonth: "$date" }, parsedDate.date()] } },
                                            { recurringYearly: true }
                                        ]
                                    }
                                ]
                            }
                        ]
                    },
                    projection: {
                        _id: 0,
                        ref: 1,
                        name: 1,
                        description: 1,
                        operational: 1,
                    }
                };
 
                const { err, result: holiday } = await holidaysHandler.read(readOptions);
 
                if (err) {
                    throw err;
                }
 
                if (holiday) {
                    operationalStatus.holiday = { ...holiday };
                }
            }
 
            return { operationalStatus };
 
        } catch (err) {
            return { err };
        }
    };
 
    methods.setInRegion = function (depots) {
 
        try {
            for (const depot of depots) {
                if (depot.type !== scenarioConst.ScenarioTypes.REGION) {
 
                    const result = depots.find(
                        x => x.childDepots && x.childDepots.find(child => child.depotRef === depot.depotRef)
                    );
 
                    depot.region = (result) ? { name: result.name, ref: result.depotRef } : null;
                } else {
                    // depot.inRegion = false;
                    depot.region = null;
                }
            }
 
            return depots;
        } catch (err) {
            console.error(err);
            return depots;
        }
    };
 
    return methods;
};
