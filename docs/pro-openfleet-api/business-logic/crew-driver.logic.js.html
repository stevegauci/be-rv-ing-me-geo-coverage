    /**
     * @method validateDepotAndAlternativeDepotChange
     * @desc Check depot and alternative depot changes, 
     * if crew's depot||alternativeDepot removed and that crew assigned the RCT & RC & DM collection with that depot->
     * block/prohibit the action
     * @param {object} options 
     * @param {string} options.collectionName
     * @param {array<object>} options.updateDatas 
     * @see methods.validateDepotsChangeMethods
     * @see methods.validateDepotsChangeMethods.constants.collectionNames for supported collections -> param options.collectionName
     * @returns { status: boolean, message: string } - { status: boolean, message: string }
     */
    methods.validateDepotAndAlternativeDepotChange = function (options = { collectionName: "", updateDatas: [] }, callback) {
 
        if (!options.collectionName || options.collectionName === "" || options.updateDatas.length === 0 || !options.updateDatas[0]) {
            return callback(new Error(l10n.t(req.headers["accept-language"], "beDepotValidationDidNotGetRequiredParameters", "Depot validation did not get required parameters")), null);
        }
 
        if (!methods.validateDepotsChangeMethods.constants.collectionNames[options.collectionName]) {
            return callback(new Error(l10n.t(req.headers["accept-language"], "beUnsupportedCollectionNameValidateDepot", "Unsupported collectionName. validateDepotAndAlternativeDepotChange()")), null);
        }
 
        const
            templateCrewOrDriverPayload = {
                limit: 5
            },//templateCrewOrDriverPayload => filter and projection dynamic
            templateRelationCollectionPayload = {
                filter: {
                    "crew": {
                        "$elemMatch": {
                            "_id": {
                                $in: []
                            }
                        }
                    }
                },
                limit: 5
            };
 
        methods.validateDepotsChangeMethods.getListCrewOrDriverByRef({ collectionName: options.collectionName, templatePayload: templateCrewOrDriverPayload, userData: options.updateDatas }, function (err, crewOrDriverList) {
            if (err) {
                return callback(err, null);
            }
 
            if (options.updateDatas.length !== crewOrDriverList.length) {
                return callback(null, { status: false, message: l10n.t(req.headers["accept-language"], "beRecordCouldNotBeFound", "Record could not be found") });
            }
 
            const {
                isSelectContainDatas,
                issuedRecordIdList,
                issuedDepotList,
            }
                = methods.validateDepotsChangeMethods.selectDepotRemovedRecords({
                    collectionName: options.collectionName,
                    userData: options.updateDatas,
                    dbData: crewOrDriverList
                });
 
            if (!isSelectContainDatas) {
                return callback(null, { status: true, message: l10n.t(req.headers["accept-language"], "beOK", "OK") });
            }
 
            methods.validateDepotsChangeMethods.getRelatedCollectionsByCrewOrDriverIdList({ idList: issuedRecordIdList, depotIdList: issuedDepotList, templatePayload: templateRelationCollectionPayload }, function (err, relatedCollectionsResult) {
                if (err) {
                    return callback(err, null);
                }
 
                const areAllEmptyListsRCTorRCorDMResult = methods.areAllEmptyListsRCTorRCorDM({
                    textStartWith: `${options.collectionName} Unlinking The Secondary Depot`,
                    rctList: relatedCollectionsResult.rctData,
                    rcList: relatedCollectionsResult.rcData,
                    dmList: relatedCollectionsResult.dmData
                });
 
                if (areAllEmptyListsRCTorRCorDMResult.status !== true) {
                    return callback(null, { status: false, message: areAllEmptyListsRCTorRCorDMResult.message });
                }
 
                return callback(null, { status: true, message: l10n.t(req.headers["accept-language"], "beOK", "OK") });
 
            });
        });
    };
 
    /**
     * @method emailValidation
     * @see emailValidationMethods
     * @desc Emails could be null, undefined, empty of string. If string contains value, 
     * it should be unique in Crews & Driver collection together. 
     * This method check that uniquness and allows other empty related values.
     * @param {object} options
     * @param {object} options.action - required emailValidationMethods.constants.actionsdb defined actions
     * @param {object} options.emailsLowerCase - required lowercased emails
     * @param {object} options.idListFromUser - UPDATE required - emailValidationMethods.actionValidationUpdate 
     * @param {object} options.refsFromUserUpperCase - BULKUPDATE required - emailValidationMethods.actionValidationBulkUpdate
     * @see CrewPersonnels - Crews 
     * @see Drivers
     * @see DeviceUsers
     * @see emailValidationMethods.constants.actionsdb
     * @returns {object} {status:Boolean , message:String}
     */
    methods.emailRefUniqueValidation = function (options, callback) {
        let { emailsLowerCase = [], crewIdList = [], driverIdList = [], refList = [] } = options,
            crewPayload,
            driverPayload,
            deviceUsersPayload,
            crewDataDbArray = [],
            driverDataDbArray = [],
            deviceUsersDataDbArray = [],
            messageOk = l10n.t(req.headers["accept-language"], "beOK", "OK");
 
        const filteredNullEmptyEmails = emailValidationMethods.filterNullEmptyEmail(emailsLowerCase);
 
        if (!emailValidationMethods.areEmailsUnique(filteredNullEmptyEmails)) {
            return { status: false, message: emailValidationMethods.getErrorMessageAlreadyExist("Payload") };
        }
 
        // Crew Payload
        crewPayload = JSON.parse(JSON.stringify(returnPayload({ emails: filteredNullEmptyEmails, usernames: refList, ids: crewIdList, usernameRef: "crewRef", idRef: "_id" })));
        crewPayload.projection = {
            "_id": 1,
            "crewRef": 1,
            "email": 1
        };
 
        // Driver Payload
        driverPayload = JSON.parse(JSON.stringify(returnPayload({ emails: filteredNullEmptyEmails, usernames: refList, ids: driverIdList, usernameRef: "driverRef", idRef: "_id" })));
        driverPayload.projection = {
            "_id": 1,
            "driverRef": 1,
            "email": 1
        };
 
        // deviceUsers Payload
        deviceUsersPayload = JSON.parse(JSON.stringify(returnPayload({ emails: filteredNullEmptyEmails, usernames: refList, ids: driverIdList, usernameRef: "username", idRef: "driver" })));
        deviceUsersPayload.projection = {
            "_id": 1,
            "username": 1,
            "email": 1,
            "driver": 1
        };
 
        //Requests
        crewHandler.list(crewPayload, function (err, crewListResult) {
            if (err) {
                return callback(err, null);
            }
 
            // Shaping the data for validation
            if (crewListResult && Array.isArray(crewListResult) && crewListResult.length > 0) {
                crewListResult.forEach(function (crewItem) {
                    if (crewItem._id) {
                        crewDataDbArray.push({
                            idString: crewItem._id.toString(),
                            email: crewItem.email ? crewItem.email.toLowerCase() : ""
                        });
                    }
                });
            }
 
            const crewValidation = actionValidation({ actionValidationData: crewDataDbArray, collectionName: "Crews", refName: "crewRef", emailsCheck: filteredNullEmptyEmails, updateId: crewIdList });
 
            if (crewValidation.status !== true) {
                return callback(null, crewValidation);
            }
 
            driverHandler.list(driverPayload, function (err, driverListResult) {
 
                if (err) {
                    return callback(err, null);
                }
 
                if (driverListResult && Array.isArray(driverListResult) && driverListResult.length > 0) {
                    driverListResult.forEach(function (driverItem) {
                        if (driverItem._id) {
                            driverDataDbArray.push({
                                idString: driverItem._id.toString(),
                                email: driverItem.email ? driverItem.email.toLowerCase() : ""
                            });
                        }
                    });
                }
 
                const driverValidation = actionValidation({ actionValidationData: driverDataDbArray, collectionName: "Drivers", refName: "driverRef", emailsCheck: filteredNullEmptyEmails, updateId: driverIdList });
 
                if (driverValidation.status !== true) {
                    return callback(null, driverValidation);
                }
 
                deviceUsersHandler.list(deviceUsersPayload, function (err, deviceUsersListResult) {
                    if (err) {
                        return callback(err, null);
                    }
 
                    // Shaping the data for validation
                    if (deviceUsersListResult && Array.isArray(deviceUsersListResult) && deviceUsersListResult.length > 0) {
                        deviceUsersListResult.forEach(function (deviceUserItem) {
                            if (deviceUserItem._id) {
                                deviceUsersDataDbArray.push({
                                    idString: deviceUserItem._id.toString(),
                                    driver: deviceUserItem.driver,
                                    email: (deviceUserItem.email) ? deviceUserItem.email.toLowerCase() : ""
                                });
                            }
                        });
                    }
 
                    const deviceUserValidation = actionValidation({ actionValidationData: deviceUsersDataDbArray, collectionName: "DeviceUsers", refName: "username", emailsCheck: filteredNullEmptyEmails, updateId: driverIdList });
 
                    if (deviceUserValidation.status !== true) {
                        return callback(null, deviceUserValidation);
                    }
 
                    return callback(null, { status: true, message: messageOk });
                });
            });
        });
    };
 
    function returnPayload(options) {
        const usernameRef = options.usernameRef;
        const idRef = options.idRef;
 
        const idCondition = options.ids ?
            {
                "$and": [
                    {
                        [usernameRef]: {
                            "$in": options.usernames
                        }
                    },
                    {
                        [idRef]: {
                            "$nin": options.ids
                        }
                    }
                ]
            } : {
                "$and": [
                    {
                        [usernameRef]: {
                            "$in": options.usernames
                        }
                    }
                ]
            };
 
        const templatePayload = {
            filter: {
                "$or": [
                    {
                        "$and": [
                            {
                                "email": {
                                    "$in": options.emails
                                }
                            },
                            {
                                [usernameRef]: {
                                    "$nin": options.usernames
                                }
                            }
                        ]
                    },
                    idCondition
                ]
            },
            projection: {},
            limit: 1
        };
 
        return templatePayload;
    }
 
    function actionValidation(options) {
        const actionValidationData = options.actionValidationData;
 
        //if actionValidationData contains some data, it means that there payload did not filter out everything, and some records do contain ref/email
        //the double check on driver field is done in case on "unconsistent" data in the collection, for ex when we have drivers without ref 
        if (actionValidationData && actionValidationData.length > 0) {
            //if email check fails
            if (options.emailsCheck && options.emailsCheck.length > 0 && actionValidationData[0].email === options.emailsCheck[0] && ((actionValidationData[0].idString !== options.updateId[0] && !actionValidationData[0].driver) || (actionValidationData[0].driver && actionValidationData[0].driver.toString() !== options.updateId[0]))) {
                return { status: false, message: emailValidationMethods.getErrorMessageEmailAlreadyExist(options.collectionName) };
            } //if ref check fails
            else if (options.updateId.length == 0 || (options.updateId.length > 0 && actionValidationData[0].idString !== options.updateId[0] && !actionValidationData[0].driver) || (actionValidationData[0].driver && actionValidationData[0].driver.toString() !== options.updateId[0])) {
                return { status: false, message: emailValidationMethods.getErrorMessageRefAlreadyExist("Ref", options.collectionName) };
            } //this is an update
            else {
                return { status: true, message: l10n.t(req.headers["accept-language"], "beOK", "OK") };
            }
        } else { //this is a create 
            return { status: true, message: l10n.t(req.headers["accept-language"], "beOK", "OK") };
        }
    }

    /**
     * @method isFieldUnchangedSingle
     * @desc checks change condition of unnested-primitive type values.
     * this methods works only with Crews and Drivers collections.
     * @param {*} objectParam 
     * @param {*} objectParam.fieldKey - db field name
     * @param {*} objectParam.collectionName - "Crews" or "Drivers"
     * @param {*} objectParam.checkValue - issued value
     * @param {*} objectParam.id - id of update element
     * @returns {status:boolean, message:string}
     */
    methods.isFieldUnchangedSingle = function ({ fieldKey, collectionName, checkValue, id }, callback) {
        let handlerLocal,
            payload,
            payloadCrew = {
                filter: {
                    "_id": id
                },
                projection: {}
            },
            payloadDriver = {
                filter: {
                    "_id": id
                },
                projection: {}
            };
        payloadCrew.projection[fieldKey] = 1;
        payloadDriver.projection[fieldKey] = 1;
 
        if (collectionName === "Crews") {
            handlerLocal = crewHandler;
            payload = payloadCrew;
 
        } else if (collectionName === "Drivers") {
            handlerLocal = driverHandler;
            payload = payloadDriver;
 
        } else {
            return callback(new Error(l10n.t(req.headers["accept-language"], "beUnsupportedCollectionName", "Unsupported collection name isUniqueRefAsync()")), null);
        }
 
        _commonRead(handlerLocal, payload, function (err, dbRecord) {
            if (err) {
                return callback(err, null);
            }
 
            if (!dbRecord) {
                return callback(null, { status: false, message: l10n.t(req.headers["accept-language"], "beOK", "Update issued record could not be found") });
            }
 
            if (!dbRecord[fieldKey] || dbRecord[fieldKey] === null || dbRecord[fieldKey] === "") {
                return callback(null, { status: true, message: l10n.t(req.headers["accept-language"], "beOK", "OK") });
            }
 
            if (checkValue.toString() === dbRecord[fieldKey].toString()) {
                return callback(null, { status: true, message: l10n.t(req.headers["accept-language"], "beOK", "OK") });
            } else {
                return callback(null, { status: false, message: l10n.t(req.headers["accept-language"], "beUpdatingProhibitedRefIsReadOnly", "Updating {1} prohibited: Ref {1} is readonly", [fieldKey]) });
            }
        });
    };
 
    /**
     * @private_commonRead
     * @desc generic method of CommonHandler.read method
     * @param {*} handler 
     * @param {*} payload 
     * @returns {callback} err/response
     */
    const _commonRead = function (handler, payload, callback) {
        handler.read(payload, function (err, response) {
            if (err) {
                return callback(err, null);
            } else {
                return callback(null, response);
            }
        });
    };

        _filterDistinct: (emailList) => {

    /**

            }

        /**
         * @desc According to Crew||Driver = idList,  alternativeDepot & Depot = depotIdList parameters
         * seeks and return the data in RCT & RC & DM collections
         * @param {object} objectParam 
         * @param {array<object>} objectParam.idList - crew||driver 's idList
         * @param {array<object>} objectParam.depotIdList - crew||driver 's changed-removed depot && alternativeDepot IdList
         * @param {object} objectParam.templatePayload - templatePayload
         * @returns { rctData:[] , rcData:[] , dmData:[]  }
         */
        getRelatedCollectionsByCrewOrDriverIdList: function ({ idList, depotIdList, templatePayload }, callback) {
 
            let RCpayload,
                RCTpayload,
                DMpayload;
 
            templatePayload.filter.crew.$elemMatch._id.$in = idList;
 
            // Payload Resource Calendar Template 
            RCTpayload = JSON.parse(JSON.stringify(templatePayload));
            RCTpayload.filter["depot"] = {
                $in: depotIdList
            };
            RCTpayload.projection = {
                depot: 1,
                dayOfWeek: 1,
                calendarTemplateRef: 1
            };
 
            // Payload Resource Calendar 
            RCpayload = JSON.parse(JSON.stringify(templatePayload));
            RCpayload.filter.depot = {
                $in: depotIdList
            };
            RCpayload.projection = {
                date: 1,
                routeRef: 1,
                depot: 1
            };
            RCpayload.populate = {
                path: "depot",
                model: Depot
            };
 
            // Payload Resource Calendar 
            DMpayload = JSON.parse(JSON.stringify(templatePayload));
            DMpayload.filter["depot._id"] = {
                $in: depotIdList
            };
            DMpayload.projection = {
                depot: 1,

            methods.commonList(resourceCalendarTemplateHandler, RCTpayload, function (err, resourceCalendarTemplateListResult) {

        /**
         * @desc Select the changed/difference/removed depot & alternativeDepot and record idlist
         * @param {object} objectParam 
         * @param {string} objectParam.collectionName
         * @param {array<object>} objectParam.userData
         * @param {array<object>} objectParam.dbData
         * @see methods.validateDepotsChangeMethods.constants.collectionNames for supported collections -> param options.collectionName
         * @returns { isSelectContainDatas:boolean, issuedRecordIdList:[], issuedDepotList:[] }
         */
        selectDepotRemovedRecords: ({ collectionName, userData, dbData }) => {
 
            let isSelectContainDatas = false,
                differenceOldDataNewData = [],
                issuedDepotList = [],
                issuedRecordIdList = [];
 
            let isValidationRequired = true;
            userData.every(i => {
                if (!i.alternativeDepots) {
                    isValidationRequired = false;
                    return false;
                }
 
                return true;
            });
 
            if (!isValidationRequired) {
                return { isSelectContainDatas, issuedRecordIdList, issuedDepotList };
            }
 
            userData.every(async itemUpdate => {
                let dbRecord;
 
                //select the correct db data
                if (collectionName === "Crews") {
                    dbRecord = _.find(dbData, dbItem => dbItem.crewRef === itemUpdate.crewRef);
 
                } else if (collectionName === "Drivers") {
                    dbRecord = _.find(dbData, dbItem => dbItem.driverRef === itemUpdate.driverRef);
                }
 
                // is there any missing alternativeDepot id on new array
                if (dbRecord.alternativeDepots && dbRecord.alternativeDepots.length > 0) {
                    dbRecord.alternativeDepots = dbRecord.alternativeDepots.map(i => i.toString());
                    differenceOldDataNewData = _.difference(dbRecord.alternativeDepots, itemUpdate.alternativeDepots);
                }
 
                //is depot changed
                if (dbRecord.depot && dbRecord.depot.toString() !== itemUpdate.depot) {
 
                    if (!issuedDepotList.includes(dbRecord.depot.toString())) {
                        issuedDepotList.push(dbRecord.depot.toString());
                    }
                    if (!issuedRecordIdList.includes(dbRecord._id.toString())) {
                        issuedRecordIdList.push(dbRecord._id.toString());
                        isSelectContainDatas = true;
                    }
                }
 
                //is one of alternative depot removed
                if (differenceOldDataNewData.length > 0) {
 
                    differenceOldDataNewData.every(itemDiffedDepotIdString => {
 
                        if (!issuedDepotList.includes(itemDiffedDepotIdString)) {
                            issuedDepotList.push(itemDiffedDepotIdString);
                        }
                        if (!issuedRecordIdList.includes(dbRecord._id.toString())) {
                            issuedRecordIdList.push(dbRecord._id.toString());
                            isSelectContainDatas = true;
                        }
                        return true;
                    });
                }
                return true;
            });
 
            return { isSelectContainDatas, issuedRecordIdList, issuedDepotList };
        },
 
        /**
         * @desc Get Crew||Driver datas for with update||bulkUpdate
         * @param {object} objectParam
         * @param {string} objectParam.collectionName
         * @param {object} objectParam.templatePayload
         * @param {array<object>} objectParam.userData
         * @see methods.validateDepotsChangeMethods.constants.collectionNames for supported collections -> param options.collectionName
         * @returns {array<object>} resultList
         */
        getListCrewOrDriverByRef: function ({ collectionName, templatePayload, userData }, callback) {
            let dataFromDb;
 
            if (collectionName === "Crews") {
 
                const refList = [],
                    payloadCrews = JSON.parse(JSON.stringify(templatePayload));
 
                userData.every(itemUserData => refList.push(itemUserData.crewRef));
 
                if (refList.length === 0 || !refList[0]) {
                    return callback(new Error(l10n.t(req.headers["accept-language"], "beCrewRefRequired", "CrewRef Required")), null);
                }
 
                payloadCrews.filter = {
                    "crewRef": { $in: refList }
                };
                payloadCrews.projection = {
                    "crewRef": 1,
                    "depot": 1,
                    "alternativeDepots": 1
                };
 
                methods.commonList(crewHandler, payloadCrews, function (err, dataFromDbResult) {
                    if (err) {
                        return callback(err, null);
                    }
 
                    dataFromDb = dataFromDbResult;
                    return callback(null, dataFromDb);
                });
            } else if (collectionName === "Drivers") {
 
                const driverRefList = [],
                    payloadDrivers = JSON.parse(JSON.stringify(templatePayload));
 
                userData.every(itemUserData => driverRefList.push(itemUserData.driverRef));
 
                if (driverRefList.length === 0 || !driverRefList[0]) {
                    return callback(new Error(l10n.t(req.headers["accept-language"], "beCrewRefRequired", "CrewRef Required")), null);
                }
 
                payloadDrivers.filter = {
                    "driverRef": { $in: driverRefList }
                };
                payloadDrivers.projection = {
                    "driverRef": 1,
                    "depot": 1,
                    "alternativeDepots": 1
                };
 
                methods.commonList(driverHandler, payloadDrivers, function (err, dataFromDbResult) {
                    if (err) {
                        return callback(err, null);
                    }
 
                    dataFromDb = dataFromDbResult;
                    return callback(null, dataFromDb);
                });
            } else {
                return callback(new Error(l10n.t(req.headers["accept-language"], "beUnsupportedCollectionForInUseDepotCheck", "Unsupported collection for in use depot check")), null);
            }
        }
    };
